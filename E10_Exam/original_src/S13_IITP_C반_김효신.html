    i = 0
    for index in str(number):
        i += int(index)
    return i

import random as r

class lotto:
    idx = int()
    num = list()

    def __init__(self, idx, num):
        self.idx = idx
        self.num = num
    def games(idx, num):
        print('일련번호', idx, '의 게임 결과는', num, '입니다.')

game = int(input('게임 수를 입력하세요. '))
for i in range(1, game+1):
    for j in range(1, game + 1):
        result = list()
        while True:
            numbers = r.randint(1, 47)
            if len(result) == 6:
                break
            else:
                result.append(numbers)
        result = list(sorted(result))
    lotto.games(i, result)

from datetime import datetime

def getMemorialDay(year, month, day, mem_day='기념일', is_msg=True):
    today = datetime.now()
    m_day = datetime(year, month, day)
    elapsed_day = today - m_day

    remain_day = abs(elapsed_day.days)
    remain_month = int(abs(remain_day / 30))
    remain_year = int(abs(remain_day / 365))

    num = list(str(remain_day))
    result = list()
    temp = ''
    n = len(num)

    for i in range(1, len(num) + 1):
        temp += num.pop(-1)
        if i % 3 == 0 and i != n:
            temp += ','
        else:
            pass
    result = temp[::-1]

    if elapsed_day.days < 0:
        if is_msg:
            print('"%s"까지 %s일 남았습니다.' % (mem_day, remain_day))
        else:
            return remain_year, remain_month, remain_day
    elif elapsed_day.days == 0:
        if is_msg:
            print('오늘입니다')
        else:
            return remain_year, remain_month, remain_day
    else:
        if is_msg:
            print(
                '오늘은 "%s"로부터 %s일 경과되었고, 달 수로는 %d개월째, 연 수로는 %d년째 되었습니다!!' % (mem_day, result, remain_month, remain_year))
        else:
            return remain_year, remain_month, remain_day
